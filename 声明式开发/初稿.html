CREATE OR REPLACE LOGIC TABLE dw_business_logic.vip_daily_summary AS SELECT u.user_id, u.username, SUM(o.pay_amount) AS daily_total, COUNT(o.order_id) AS order_cnt FROM ods_raw_data.users_info u -- ← 指定不同所在的 Database LEFT JOIN ods_raw_data.orders_raw o -- ← 同连接内跨库关联 ON u.user_id = o.user_id AND DATE(o.create_time) = CURDATE() WHERE u.status = 1 GROUP BY u.user_id, u.username;
